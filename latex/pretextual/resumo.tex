\begin{resumo}
	\fixme{Este trabalho tem como objetivo apresentar os conceitos de estruturas de dados por meio de um jogo sério. O projeto busca facilitar o aprendizado dos alunos de forma lúdica e interativa. A metodologia utilizada foi baseada no desenvolvimento ágil, com foco em testes práticos. Os resultados demonstraram que a gamificação pode ser uma ferramenta eficaz no ensino de lógica e programação.}

	\vspace{\onelineskip}
	\noindent
	\textbf{Palavras-chave}: Jogo Sério. Estruturas de Dados. Ensino. Gamificação.

\end{resumo}
